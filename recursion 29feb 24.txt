1. Write a recursive function to print numbers from 1 to 15.

#include <iostream>

using namespace std;
void repetition(int i){
    if(i<1)
    return;
    
    repetition(i-1);
    cout<<i<<" "<<"\n";
}

int main()
{
    int i=15;
    repetition(i);

    return 0;
}
